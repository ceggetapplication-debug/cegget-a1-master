        return {

const handlePlaceOrderClick = () => {
    const fullName = userLocation!.fullName || '';
    let nameInitials = '';
    const nameParts = fullName.split(' ').filter(part => part.length > 0);
    if (nameParts.length > 0) {
        nameInitials += nameParts[0].charAt(0).toUpperCase();
    }
    if (nameParts.length > 1) {
        nameInitials += nameParts[nameParts.length - 1].charAt(0).toUpperCase();
    }

    const username = userLocation!.username || '';
    const usernamePart = username.length > 0 ? username.substring(0, Math.min(username.length, 3)).toUpperCase() : '';
    let userIdentifier = `${nameInitials}${usernamePart}`;
    if (userIdentifier.length === 0) userIdentifier = 'USER';

    const today = new Date();
    const day = String(today.getDate()).padStart(2, '0');
    const month = String(today.getMonth() + 1).padStart(2, '0');
    const year = today.getFullYear();
    const formattedDate = `${day}${month}${year}`;

    const ref = `${userIdentifier}-${formattedDate}-${userOrderCount}`;
    setOrderReference(ref);

    const emailData = buildOrderSummaryEmailData();
    setOrderSummaryEmailData(emailData);

    setShowOrderSummary(true);
};


const handleConfirmOrderSummary = () => {

    if (selectedProductsInSummary.length === 0) {
        return <Text style={styles.emptyCartText}>{t('no_products_selected')}</Text>;
    }

    return selectedProductsInSummary.map(product => {
        const store = stores.find(s => s.id === product.storeId);
        const itemPrice = (product.valeurQuantite ?? 0) * product.prix;
        return (

                <Text style={styles.summaryProductName}>{product.nom} - {product.marque}</Text>
                <Text style={styles.summaryProductDetails}>

                <Text style={styles.summaryProductPrice}>
                    {itemPrice.toFixed(2)} DZD

    });
};
